{- |
Flag to change behavior of `dockerBuildImage`
-}
data Command
  = BuildOnly -- ^ Only build Docker images
  | BuildAndPush -- ^ Build and push to AWS


{- |
Flag to change behavior of `dockerBuild`
-}
data BuildMode
  = BuildAll
  | BuildUpdated


-- | Get build path (all paths are relative to the root directory)
buildDir :: String -> String
buildDir path =
  "./build" ++ path


-- | Get source path (all paths are relative to the root directory)
sourceDir :: String -> String
sourceDir path =
  "./source" ++ path


removeDirectories :: String -> Action ()
removeDirectories name = do
  putNormal $ "Remove " ++ name ++ " directories:"
  indexFiles <- getDirectoryFiles "" ["//*/" ++ name ++ "/index.js"]
  let dirs = map takeDirectory indexFiles
  void $ forP dirs (\dir -> removeFilesAfter dir ["//*"])

-- * Utility Functions

-- | Delete all build artifacts and files from shake's cache
clean :: Action ()
clean = do
  putNormal "Remove files in shakecache"
  removeFilesAfter "shakecache" ["//*"]

  putNormal "Remove files in build directory"
  removeFilesAfter "build" ["//*"]


-- | Upgrade all Modules
upgrade :: Action ()
upgrade = do
  lockFiles <- getDirectoryFiles "" ["build/*/yarn.lock"]
  -- Should not be implemented with `need`
  -- as `yarn` might accidentally be executed and rewrite `yarn.lock` files
  -- Must be run sequentially due to yarn's bad support for parallelization
  void $ forM lockFiles yarnUpgrade


-- | Upgrade yarn dependencies (changes `yarn.lock` file)
yarnUpgrade :: FilePath -> Action ()
yarnUpgrade out = do
  let directory = takeDirectory out
  -- Needing readme leads to package.json being copied into the directory
  need [directory ++ "/readme.md"]
  cmd_ (Cwd directory) "yarn upgrade"


-- | Upgrade api-client
upgradeApiClient :: Action ()
upgradeApiClient = do
  lockFiles <- getDirectoryFiles "" ["build/*/yarn.lock"]
  void $ forM lockFiles (\out -> do
      let directory = takeDirectory out
      need [directory ++ "/readme.md"]
      cmd_ (Cwd directory)
        "yarn add git+ssh://git@github.com/feramhq/api-client#master"
    )


startsWithOneOf :: [String] -> String -> Bool
startsWithOneOf prefixes value =
  any (`isPrefixOf` value) prefixes


isModule :: String -> Bool
isModule =
  startsWithOneOf ["fix-", "warn-"]


wrapMessage :: String -> String
wrapMessage message =
  let line = "==================================================\n"
  in line ++ message ++ "\n" ++ line


messageForceBuild :: [String] -> String
messageForceBuild modules = wrapMessage $ ""
  ++ "Force rebuild of all images:\n"
  ++  (intercalate ", " modules)


messageNoChange :: String -> [String] -> String
messageNoChange commitRange modules = wrapMessage $ ""
  ++ "Commits " ++ commitRange ++ " changed no modules\n"
  ++ "-> Build all images:\n"
  ++  (intercalate ", " modules)


messageChange :: String -> [String] -> String
messageChange commitRange changedDirs = wrapMessage $ ""
  ++ "Commits " ++ commitRange ++ " changed following modules:\n"
  ++  (intercalate ", " changedDirs)


getDirsFromDiff :: String -> [String]
getDirsFromDiff diffString = diffString
  & lines
  & filter (takeDirectory >>> (/= "."))
  & map takeDirectory1
  & nub -- Strange Haskell name for uniq function
  & filter isModule


{- |
Convert a possibly empty foldable to a maybe
> emptyToMaybe "" == Nothing
> emptyToMaybe "text" == Just "text"
-}
emptyToMaybe :: Foldable t => t a -> Maybe (t a)
emptyToMaybe value =
  if null value
  then Nothing
  else Just value


{- |
Convert a possibly not existant and/or empty `Foldable` to a `Maybe`
> ensureNonEmpty Nothing == Nothing
> ensureNonEmpty (Just "") == Nothing
> ensureNonEmpty (Just "text") == Just "text"
-}
ensureNonEmpty :: (Foldable t) => Maybe (t a) -> Maybe (t a)
ensureNonEmpty value =
  fromMaybe Nothing (fmap emptyToMaybe value)


-- * Build Targets

{- |
Build and push a single docker image
-}
dockerBuildImage :: Command -> String -> String -> Action ()
dockerBuildImage buildCommand sshKey dir = do
  let
    name = takeBaseName dir
    message = "\n\n===== Run `docker build` in \"" ++ dir ++ "\" ====="

  putNormal message

  Stdout gitSha <- (cmd "git rev-parse HEAD")
  travisBranch <- getEnv "TRAVIS_BRANCH"
  travisBranchPR <- getEnv "TRAVIS_PULL_REQUEST_BRANCH"

  let
    gitShaClean = trim (gitSha :: String)
    branch = fromMaybe "HEAD"
      ( ensureNonEmpty travisBranchPR
        <|> ensureNonEmpty travisBranch
        <|> ensureNonEmpty (Just gitShaClean)
      )
    awsPrefix = "799797707255.dkr.ecr.us-east-1.amazonaws.com/modules/"
    nameWithBranch = awsPrefix ++ name ++ ":" ++ branch
    nameWithSha = awsPrefix ++ name ++ ":" ++ gitShaClean

  cmd_
    (AddEnv "SSH_KEY" sshKey)
    (Cwd dir)
    "docker build"
    "--build-arg SSH_KEY"  -- implicitly means SSH_KEY="$SSH_KEY"
    "--tag" [nameWithBranch]
    "--tag" [nameWithSha]
    ["."]

  case buildCommand of
    BuildOnly -> do
      putNormal "Docker image was built"

    BuildAndPush -> do
      cmd_ "docker push" [nameWithBranch]
      cmd_ "docker push" [nameWithSha]
      putNormal "Docker image was built and pushed"


{- |
Build docker images of changed modules
or all modules when other files were changed.
-}
dockerBuild :: Command -> BuildMode -> Action ()
dockerBuild buildCommand mode = do

  -- Define dependencies on environment variables
  envSshKey <- getEnv "SSH_KEY"

  -- TODO: Rewrite with MaybeT
  case envSshKey of
    Nothing -> do
      putQuiet "Error: SSH_KEY must be set"

    Just sshKey -> do
      let
        modulesDir = "source/tools"

      dirs <- getDirectoryDirs modulesDir

      let
        nameToReadme name = buildDir $ "/" ++ name ++ "/readme.md"
        nameToAnalytics name =
          buildDir $ "/" ++ name ++ "/analytics/package.json"
        nameToBuildPatch name =
          buildDir $ "/" ++ name ++ "/build-patch/package.json"

      need $ fmap nameToReadme dirs
      need $ fmap nameToAnalytics dirs  -- TODO: Only build if needed
      need $ fmap nameToBuildPatch dirs  -- TODO: Only build if needed

      envCommitRange <- getEnvWithDefault
        "HEAD~1..HEAD"
        "TRAVIS_COMMIT_RANGE"
      -- TODO: Increase robustness in face of history rewrites
      --   (as seen in https://github.com/nest/nest-simulator/pull/75/files)
      Stdout diffString <-
        let gitCmd = "git diff --relative --name-only"
        in cmd (Cwd modulesDir) gitCmd [envCommitRange]

      let
        changedDirs = getDirsFromDiff diffString
        prefixModules = (buildDir "/" ++)
        modules = fmap prefixModules dirs
        changedModules = fmap prefixModules changedDirs

      -- Get command to login to AWS with Docker
      Stdout dockerCommand <- cmd "\
        \aws ecr get-login \
        \--region us-east-1 \
        \--no-include-email \
        \"

      -- Login to AWS
      cmd_ (dockerCommand :: String)

      let
        (modulesToBuild, message) = case mode of
          BuildAll ->
            (modules, messageForceBuild modules)
          BuildUpdated ->
            if null changedDirs
            then (modules, messageNoChange envCommitRange modules)
            else (changedModules, messageChange envCommitRange changedDirs)

      putNormal message

      void $ forP modulesToBuild (dockerBuildImage buildCommand sshKey)


-- | Generate documentation for Haskell code
buildDocs :: Action ()
buildDocs = do
  cmd_
    "stack exec --"
    "haddock"
      "--html"
      "--hyperlinked-source"
      "--odir" "docs"
      "Modules/Shakefile.hs"


customShakeOptions :: ShakeOptions
customShakeOptions = shakeOptions
  { shakeColor = True  -- Print colors to terminal
  , shakeThreads = 0  -- Match number of processors
  , shakeTimings = True  -- Print runtime statistics
  , shakeFiles = "shakecache" -- Save cache files to `shakecache`
  -- Enable following line if you need to check out the report
  -- , shakeReport = ["shake-report.html"] -- Enable this
  }


-- | List all actions and call `dockerBuild` with `BuildOnly` per default
main :: IO ()
main = shakeArgs customShakeOptions $ do

  -- Specify build rules for Feram modules
  -- (These specific rules have higher priority than the wildcard rules)
  Autopep8.build
  Bandit.build
  Brakeman.build
  Eslint.build
  Govet.build
  Hadolint.build
  Jsonlint.build
  PackageFile.build
  PhpCsFixer.build
  PhpSecurityChecker.build
  Retirejs.build
  Rubocop.build
  Shellcheck.build
  Stylefmt.build
  Stylelint.build
  Svgo.build
  TargetBlank.build
  Textlint.build
  Tidy.build
  TrailingNewline.build
  Tslint.build
  Typos.build
  Uncrustify.build
  Yamllint.build

  -- Specify build rules for EcmaScript module dependencies
  Analytics.build
  BuildPatch.build


  "build//readme.md" %> \out -> do
    let srcDir = (replaceDirectory (takeDirectory out) "source/tools") ++ "/"
    cmd_ "rsync --recursive --update --links" srcDir (dropFileName out)

  -- Upgrade all modules with yarn upgrade
  phony "upgrade" upgrade

  -- Upgrade api-client
  phony "upgrade-api-client" upgradeApiClient


  -- Build a single container
  phony "build-single" $ do
    let moduleName = "fix-package-file"

    need $ fmap (buildDir . ("/" ++)) (
      (moduleName ++ "/readme.md") :
      (moduleName ++ "/analytics/package.json") :
      (moduleName ++ "/build-patch/package.json") :
      [])

    envSshKey <- getEnv "SSH_KEY"
    case envSshKey of
      Nothing -> putNormal "SSH_KEY is not set"
      Just key ->
        dockerBuildImage BuildOnly key (buildDir "/" ++ moduleName)


  -- Run unit tests of all modules
  phony "test" $ do
    dirs <- getDirectoryDirs "source/tools"

    let
      nameToReadme name = buildDir $ "/" ++ name ++ "/readme.md"
      toolReadmes = fmap nameToReadme dirs
      modulesToBuild = fmap (buildDir "/" ++) dirs

    need toolReadmes
    void $ forM modulesToBuild (\moduleDir -> do
        putNormal $ "=====\n" ++ moduleDir
          ++ ": Install dependencies and run tests"
        cmd_ (Cwd moduleDir) "yarn"
        cmd_ (Cwd moduleDir) "yarn test"
      )


  phony "build-docker-images" $ dockerBuild BuildOnly BuildUpdated
  phony "build-all-docker-images" $ dockerBuild BuildOnly BuildAll

  phony "build-push-docker-images" $ dockerBuild BuildAndPush BuildUpdated
  phony "build-push-all-docker-images" $ dockerBuild BuildAndPush BuildAll

  phony "build-docs" buildDocs

  phony "clean" clean


  want ["build-docker-images"]
