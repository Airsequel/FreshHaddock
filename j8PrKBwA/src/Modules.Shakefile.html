{- |
Flag to change behavior of `dockerBuildImage`
-}
data Command
  = BuildOnly -- ^ Only build Docker images
  | BuildAndPush -- ^ Build and push to AWS


-- * Utility Functions

-- | Delete all files from shake's cache
clean :: Action ()
clean = do
  putNormal "Cleaning files in shakecache"
  removeFilesAfter "shakecache" ["//*"]


-- | Upgrade all Modules
upgrade :: Action ()
upgrade = do
  packageFiles <- getDirectoryFiles "" ["*/package.json"]
  let lockFiles = map ((</> "yarn.lock") . takeDirectory) packageFiles
  need lockFiles


-- | Upgrade yarn dependencies (changes `yarn.lock` file)
yarnUpgrade :: FilePath -> Action ()
yarnUpgrade out = do
  let directory = takeDirectory out
  let command = "yarn upgrade"
  cmd (Cwd directory) command


startsWithOneOf :: [String] -> String -> Bool
startsWithOneOf prefixes value =
  any (`isPrefixOf` value) prefixes


isModule :: String -> Bool
isModule =
  startsWithOneOf ["fix-", "warn-"]


messageNoChange :: String -> [String] -> String
messageNoChange commitRange modules = ""
  ++ "==================================================\n"
  ++ "Commits " ++ commitRange ++ " changed no modules  \n"
  ++ "-> Build all images:                              \n"
  ++  (intercalate ", " modules) ++ "                   \n"
  ++ "==================================================\n"


messageChange :: String -> [String] -> String
messageChange commitRange changedDirs = ""
  ++ "==================================================\n"
  ++ "Commits " ++ commitRange ++ " changed following modules:\n"
  ++  (intercalate ", " changedDirs) ++ "               \n"
  ++ "==================================================\n"


getDirsFromDiff :: String -> [String]
getDirsFromDiff diffString = words diffString
  & filter (takeDirectory >>> (/= "."))
  & map takeDirectory1
  & nub -- Strange Haskell name for uniq function
  & filter isModule


{- |
Convert a possibly empty foldable to a maybe
> emptyToMaybe "" == Nothing
> emptyToMaybe "text" == Just "text"
-}
emptyToMaybe :: Foldable t => t a -> Maybe (t a)
emptyToMaybe value =
  if null value
  then Nothing
  else Just value


{- |
Convert a possibly not existant and/or empty `Foldable` to a `Maybe`
> ensureNonEmpty Nothing == Nothing
> ensureNonEmpty (Just "") == Nothing
> ensureNonEmpty (Just "text") == Just "text"
-}
ensureNonEmpty :: (Foldable t) => Maybe (t a) -> Maybe (t a)
ensureNonEmpty value =
  fromMaybe Nothing (fmap emptyToMaybe value)


-- * Build Targets

{- |
Build of build & push a single docker image
-}
dockerBuildImage :: Command -> String -> String -> Action ()
dockerBuildImage command sshKey dir = do
  putNormal $ "\n\n===== Run `docker build` in \"" ++ dir ++ "\" ====="

  Stdout gitSha <- (cmd "git rev-parse HEAD")
  travisBranch <- getEnv "TRAVIS_BRANCH"
  travisBranchPR <- getEnv "TRAVIS_PULL_REQUEST_BRANCH"

  let
    gitShaClean = trim (gitSha :: String)
    branch = fromMaybe "HEAD"
      ( ensureNonEmpty travisBranchPR
        <|> ensureNonEmpty travisBranch
        <|> ensureNonEmpty (Just gitShaClean)
      )
    awsPrefix = "799797707255.dkr.ecr.us-east-1.amazonaws.com/modules/"
    nameWithBranch = awsPrefix ++ dir ++ ":" ++ branch
    nameWithSha = awsPrefix ++ dir ++ ":" ++ gitShaClean

  cmd_ (Cwd dir) "rm -rf ./build-patch"
  cmd_ (Cwd dir) "cp -r ../build-patch/ ./build-patch"
  cmd_
    (AddEnv "SSH_KEY" sshKey)
    (Cwd dir)
    "docker build"
    "--build-arg SSH_KEY"  -- implicitly means SSH_KEY="$SSH_KEY"
    "--tag" [nameWithBranch]
    "--tag" [nameWithSha]
    ["."]

  case command of
    BuildOnly -> do
      putNormal "Docker image was built"

    BuildAndPush -> do
      cmd_ "docker push" [nameWithBranch]
      cmd_ "docker push" [nameWithSha]
      putNormal "Docker image was built and pushed"


{- |
Build docker images of changed modules
or all modules when other files were changed.
-}
dockerBuild :: Command -> Action ()
dockerBuild command = do
  -- Define dependencies on environment variables
  envSshKey <- getEnv "SSH_KEY"

  -- TODO: Rewrite with MaybeT
  case envSshKey of
    Nothing -> do
      putQuiet "Error: SSH_KEY must be set"

    Just sshKey -> do
      envCommitRange <- getEnvWithDefault
        "HEAD~1..HEAD"
        "TRAVIS_COMMIT_RANGE"
      -- TODO: Increase robustness in face of history rewrites
      --   (as seen in https://github.com/nest/nest-simulator/pull/75/files)
      Stdout diffString <- cmd "git diff --name-only" [envCommitRange]

      let changedDirs = getDirsFromDiff diffString

      -- Get command to login to AWS with Docker
      Stdout dockerCommand <- cmd "\
        \aws ecr get-login \
        \--region us-east-1 \
        \--no-include-email \
        \"

      -- Login to AWS
      cmd_ (dockerCommand :: String)

      dirs <- getDirectoryDirs ""

      let
        (modulesToBuild, message) = if null changedDirs
          then
            let modules = dirs & filter isModule
            in (modules, messageNoChange envCommitRange modules)
          else
            (changedDirs, messageChange envCommitRange changedDirs)

      putNormal message

      void $ forP modulesToBuild (dockerBuildImage command sshKey)


-- | Generate documentation for Haskell code
buildDocs :: Action ()
buildDocs = do
  cmd_
    "stack exec --"
    "haddock"
      "--html"
      "--hyperlinked-source"
      "--odir" "docs"
      "Modules/Shakefile.hs"


customShakeOptions :: ShakeOptions
customShakeOptions = shakeOptions
  { shakeColor = True  -- Print colors to terminal
  , shakeThreads = 0  -- Match number of processors
  , shakeTimings = True  -- Print runtime statistics
  , shakeFiles = "shakecache" -- Save cache files to `shakecache`
  -- Enable following line if you need to check out the report
  -- , shakeReport = ["shake-report.html"] -- Enable this
  }


-- | List all actions and call `dockerBuild` with `BuildOnly` per default
main :: IO ()
main = shakeArgs customShakeOptions $ do
  want ["build-docker-images"]

  -- Upgrade all modules with yarn upgrade
  phony "upgrade" upgrade

  "//yarn.lock" %> yarnUpgrade


  -- Build and push single containers
  -- TODO: Refactor and remove code duplication
  phony "build-single" $ do
    let name = "warn-hadolint"
    envSshKey <- getEnv "SSH_KEY"

    Stdout dockerCommand <- cmd "\
        \aws ecr get-login \
        \--region us-east-1 \
        \--no-include-email \
        \"

    -- Login to AWS
    cmd_ (dockerCommand :: String)

    case envSshKey of
      Nothing -> putNormal "SSH_KEY is not set"
      Just key -> dockerBuildImage BuildAndPush key name


  phony "build-docker-images" $ dockerBuild BuildOnly

  phony "build-push-docker-images" $ dockerBuild BuildAndPush

  phony "build-docs" buildDocs

  phony "clean" clean
