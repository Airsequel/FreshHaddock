{-# LANGUAGE DeriveAnyClass #-}
{-# LANGUAGE DeriveGeneric #-}
{-# LANGUAGE FlexibleContexts #-}
{-# LANGUAGE NoImplicitPrelude #-}
{-# LANGUAGE OverloadedLists #-}
{-# LANGUAGE OverloadedStrings #-}
{-# LANGUAGE RecordWildCards #-}


module Modules.Tool where

import Protolude hiding (link)
import GHC.Generics (Generic)
import Data.Aeson
import qualified Data.Text as Text


data Language
  = C
  | CPlusPlus
  | CSharp
  | CSS
  | D
  | Dockerfile
  | Go
  | HTML
  | Java
  | JavaScript
  | JSON
  | LanguageIndependent
  | Less
  | Markdown
  | Npm
  | ObjectiveC
  | Pawn
  | PHP
  | PlainText
  | Python
  | Ruby
  | SCSS
  | ShellScript
  | SugarSS
  | SVG
  | TypeScript
  | VALA
  | XML
  | YAML
  deriving (Eq, Show, Generic)

instance ToJSON Language where
  toJSON CPlusPlus = String "C++"
  toJSON CSharp = String "C#"
  toJSON ObjectiveC = String "Objective-C"
  toJSON PlainText = String "Plain Text"
  toJSON ShellScript = String "Shell Script"
  toJSON LanguageIndependent = String "Language Independent"
  toJSON language = String (Text.pack $ show language)


data Tool = Tool
  { id :: Text
  , name :: Text
  , version :: Text
  , description :: Text
  , link :: Text
  , languages :: [Language]
  } deriving (Eq, Show, Generic, ToJSON)
