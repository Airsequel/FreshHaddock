copyFrom :: MonadFree EInstruction m
  => [Char]
  -> (NonEmpty SourcePath)
  -> TargetPath
  -> m ()
copyFrom image sources dest =
  copyArgs sources dest Syntax.NoChown (Syntax.CopySource image)


getDockerfile :: [Char]
getDockerfile = toDockerfileStr $ do
  let
    tag = "v1-11-gdf132ae_2017-11-02"
    imageName = "ebekebe/ferambase"
    alias = "builder"
    healthCheckPort = 5000 :: Integer

  from $ (tagged imageName tag) `aliased` alias

  workdir "/var/app"

  arg "SSH_KEY" Nothing
  runArgs ["configure-ssh", "$SSH_KEY"]

  copy ["package.json"] "package.json"
  copy ["yarn.lock"] "yarn.lock"
  copy ["build-patch"] "build-patch"

  runArgs ["yarn", "install"]

  ------------------------------

  from $ tagged "node" "8-slim"
  workdir "/var/app"

  copyFrom alias ["/var/app/node_modules"] "node_modules"

  run "apt-get update --yes"
  run "apt-get upgrade --yes"
  run "apt-get install ruby-full --yes"
  run $ T.unpack $ "gem install rubocop --version " <> toolVersion

  copy ["yarn.lock"] "yarn.lock"
  copy ["package.json"] "package.json"
  copy ["tests"] "tests"
  copy ["source"] "source"

  runArgs ["yarn", "test"]

  -- TODO: Improve after https://github.com/hadolint/language-docker/issues/22
  entrypoint $ show
    (["yarn", "start", "--health-check-port", show healthCheckPort] :: [[Char]])

  healthcheck $
    (check $ "curl -f http://localhost:" ++ show healthCheckPort
      ++ " > /dev/null || exit 1")
    `EDSL.interval` 10 {-sec-}
    `EDSL.timeout` 3 {-sec-}

getConfigFile :: [Char]
getConfigFile = do
  Config.showPretty ModuleConfig
    { confName = name
    , confVersion = version
    , confDescription = description
    , confType = FixingFunction
    , confSignature = Signature
      { sigType = JsObject
      , sigProperties = properties
        [ ("absoluteFilePath", JsString)
        , ("fileName", JsString)
        , ("text", JsString)
        ]
      , sigReturnType = JsPatchObjects
      }
    , confTool = Tool.Tool
      { Tool.id = toolId
      , Tool.name = toolName
      , Tool.version = toolVersion
      , Tool.description = description
      , Tool.link = "https://github.com/bbatsov/rubocop"
      , Tool.languages = [Ruby]
      }
    , confConfigFiles = []
    , confFilePattern = JsRegex [IgnoreCase] "\\.rbw?$"
    , confFileType = HumanReadable
    }


build :: Rules ()
build = do
  let
    dockerfile = buildDir id "/Dockerfile"
    packgeJson = buildDir id "/package.json"
    yarnLock = buildDir id "/yarn.lock"
    dockerignore = buildDir id "/.dockerignore"
    configJs = buildDir id "/source/config.json"
    indexJs = buildDir id "/source/index.js"
    mainJs = buildDir id "/source/main.js"
    testJs = buildDir id "/tests/main.js"

    flawedRb = buildDir id "/tests/fixtures/flawed.rb"
    flawedUnfixableRb = buildDir id "/tests/fixtures/flawed-unfixable.rb"
    correctRb = buildDir id "/tests/fixtures/correct.rb"


  -- Proxy file for whole module
  buildDir id "/readme.md" %> \out -> do
    need (
      dockerfile :
      packgeJson :
      yarnLock :
      dockerignore :
      configJs :
      indexJs :
      mainJs :
      testJs :
      flawedRb :
      flawedUnfixableRb :
      correctRb :
      [])
    writeFileChanged out "⚠ This directory contains only build files ⚠"


  packgeJson %> \out -> do
    writeFileChanged out getPackageFile


  buildDir id "/source/*.js" %> \out -> do
    copyFileChanged ((sourceDir id "/source/") <> (takeFileName out)) out


  buildDir id "/source/config.json" %> \out -> do
    writeFileChanged out getConfigFile


  testJs %> \out -> do
    copyFileChanged (replace "build" "source/tools" out) out


  dockerfile %> \out -> do
    writeFileChanged out getDockerfile


  dockerignore %> \out -> do
    writeFileChanged out "/node_modules/**\n"


  flawedRb %> \out -> do
    copyFileChanged (replace "build" "source/tools" out) out


  flawedUnfixableRb %> \out -> do
    writeFileChanged out "testArray = %w[T E S T]"


  correctRb %> \out -> do
    copyFileChanged (replace "build" "source/tools" out) out
