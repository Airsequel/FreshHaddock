{-# LANGUAGE OverloadedStrings #-}
{-# LANGUAGE OverloadedLists #-}
{-# LANGUAGE FlexibleContexts #-}


module Modules.FixPackageFile where

import Development.Shake
import Development.Shake.FilePath

import Modules.PackageJson
import Modules.DockerfileGenerator


sourceDir :: String -> String
sourceDir path =
  "./source/tools/fix-package-file" ++ path


buildDir :: String -> String
buildDir path =
  "./build/fix-package-file" ++ path

build :: Rules ()
build = do
  -- Proxy file for whole module
  buildDir "/readme.md" %> \out -> do
    need (
      (buildDir "/Dockerfile") :
      (buildDir "/package.json") :
      (buildDir "/yarn.lock") :
      (buildDir "/.dockerignore") :
      (buildDir "/source/config.js") :
      (buildDir "/source/index.js") :
      (buildDir "/source/main.js") :
      (buildDir "/tests/main.js") :
      (buildDir "/tests/repos/basic/package.json") :
      (buildDir "/tests/repos/basic/.git/HEAD") :
      (buildDir "/tests/repos/no-package-file/readme.md") :
      (buildDir "/tests/repos/no-package-file/.git/HEAD") :
      [])
    writeFileChanged out "⚠ This directory contains only build files ⚠"


  buildDir "/Dockerfile" %> \out -> do
    writeFileChanged out (makeDockerfile $ return ())


  buildDir "/source/*.js" %> \out -> do
    copyFileChanged ((sourceDir "/source/") ++ (takeFileName out)) out


  buildDir "/tests/main.js" %> \out -> do
    copyFileChanged (sourceDir "/tests/main.js") out


  buildDir "/package.json" %> \out -> do
    writeFileChanged out getPackageFile


  buildDir "/.dockerignore" %> \out -> do
    writeFileChanged out "/node_modules/**\n"


  buildDir "/tests/repos/basic/package.json" %> \out -> do
    writeFileChanged out "\
      \{\"name\":\"broken-files\",\"version\":\"0.1.0\",\n\
      \\"description\":\"This repo contains several broken files for testing purposes\",\n\
      \\"repository\":{\"type\":\"git\",\n\
      \  \"url\":\"git+https://github.com/ferambot/broken.git\"},\n\
      \\"author\":\"Adrian Sieber\",\"license\":\"ISC\",\n\
      \\"bugs\": {\"url\":\"https://github.com/ferambot/broken/unavailable-website\"},\n\
      \\"homepage\":\"https://github.com/ferambot/broken\"}\n\
      \"


  buildDir "/tests/repos/basic/.git/HEAD" %> \out -> do
    writeFileChanged out "ref: refs/heads/master\n"


  buildDir "/tests/repos/no-package-file/readme.md" %> \out -> do
    writeFileChanged out "Just a placeholder\n"


  buildDir "/tests/repos/no-package-file/.git/HEAD" %> \out -> do
    writeFileChanged out "ref: refs/heads/master\n"
