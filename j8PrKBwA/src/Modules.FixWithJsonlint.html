sourceDir :: String -> String
sourceDir path =
  "./source/tools/fix-with-jsonlint" ++ path


buildDir :: String -> String
buildDir path =
  "./build/fix-with-jsonlint" ++ path


copyFrom :: MonadFree EInstruction m
  => String
  -> (NonEmpty SourcePath)
  -> TargetPath
  -> m ()
copyFrom image sources dest =
  copyArgs sources dest Syntax.NoChown (Syntax.CopySource image)


getDockerfile :: String
getDockerfile = toDockerfileStr $ do
  let
    tag = "v1-11-gdf132ae_2017-11-02"
    name = "ebekebe/ferambase"
    alias = "builder"

  from $ (tagged name tag) `aliased` alias

  workdir "/var/app"

  arg "SSH_KEY" Nothing
  runArgs ["configure-ssh", "$SSH_KEY"]

  copy ["package.json"] "package.json"
  copy ["yarn.lock"] "yarn.lock"
  copy ["build-patch"] "build-patch"

  runArgs ["yarn", "install"]

  ------------------------------

  from "node:8-alpine"
  workdir "/var/app"

  copyFrom alias ["/var/app/node_modules"] "node_modules"

  copy ["yarn.lock"] "yarn.lock"
  copy ["package.json"] "package.json"
  copy ["tests"] "tests"
  copy ["source"] "source"

  runArgs ["yarn", "test"]

  -- TODO: Fix after https://github.com/hadolint/language-docker/issues/22
  entrypoint $ show (["yarn", "start", "--"] :: [String])

build :: Rules ()
build = do
  -- Proxy file for whole module
  buildDir "/readme.md" %> \out -> do
    need (
      (buildDir "/Dockerfile") :
      (buildDir "/package.json") :
      (buildDir "/yarn.lock") :
      (buildDir "/.dockerignore") :
      (buildDir "/source/config.js") :
      (buildDir "/source/index.js") :
      (buildDir "/source/main.js") :
      (buildDir "/tests/main.js") :
      (buildDir "/tests/fixtures/correct.json") :
      (buildDir "/tests/fixtures/wrong-indentation.json") :
      (buildDir "/tests/fixtures/missing-comma.json") :
      [])
    writeFileChanged out "⚠ This directory contains only build files ⚠"


  buildDir "/Dockerfile" %> \out -> do
    writeFileChanged out getDockerfile


  buildDir "/source/*.js" %> \out -> do
    copyFileChanged ((sourceDir "/source/") ++ (takeFileName out)) out


  buildDir "/tests/main.js" %> \out -> do
    copyFileChanged (sourceDir "/tests/main.js") out


  buildDir "/package.json" %> \out -> do
    writeFileChanged out getPackageFile


  buildDir "/.dockerignore" %> \out -> do
    writeFileChanged out "/node_modules/**\n"


  buildDir "/tests/fixtures/correct.json" %> \out -> do
    writeFileChanged out "\
      \{\n\
      \  \"great\": 13,\n\
      \  \"code\": 37\n\
      \}\n\
      \"


  buildDir "/tests/fixtures/wrong-indentation.json" %> \out -> do
    writeFileChanged out "\
      \ { \"key\":     \"value \" ,\n\
      \\"another key\"  :20}\n\
      \"


  buildDir "/tests/fixtures/missing-comma.json" %> \out -> do
    writeFileChanged out "\
      \{\n\
      \  \"key\": \"value\"\n\
      \  \"another key\": \"another value\"\n\
      \}\n\
      \"
