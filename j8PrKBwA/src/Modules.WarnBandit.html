id, name, version, description, toolId, toolName, toolVersion :: Text

id = "warn-bandit"
name = "Warn Bandit"
version = "1.1.0"
description = "Perform security analysis on Python source code"

toolId = "bandit"
toolName = "Bandit"
toolVersion = "1.4.0"


getPackageFile :: [Char]
getPackageFile = do
  PackJson.showPretty packageFileDefault
    { packName = "@feram/" <> id
    , packVersion = version
    , packDescription = description
    , packDependencies = dependencies (
        ( "@feram/api-client", "master",
          Just "git+ssh://git@github.com/feramhq/api-client") :
        ( "@feram/build-patch", "", Just "./build-patch") :
        ("bunyan", "^1.8.5", Nothing) :
        ("execa", "^0.6.3", Nothing) :
        ("fs-extra", "^5.0.0", Nothing) :
        [])
    }


getConfigFile :: [Char]
getConfigFile = do
  Config.showPretty ModuleConfig
    { confName = name
    , confVersion = version
    , confDescription = description
    , confType = FixingFunction
    , confSignature = Signature
      { sigType = JsObject
      , sigProperties = properties
        [ ("absoluteFilePath", JsString)
        , ("fileName", JsString)
        , ("repoPath", JsString)
        , ("text", JsString)
        ]
      , sigReturnType = JsPatchObjects
      }
    , confTool = Tool.Tool
      { Tool.id = toolId
      , Tool.name = toolName
      , Tool.version = toolVersion
      , Tool.description = description
      , Tool.link = "https://github.com/openstack/bandit"
      , Tool.languages = [Python]
      }
    , confConfigFiles =
      [ ".editorconfig"
      , "tox.ini"
      ]
    , confFilePattern = JsRegex [IgnoreCase] "\\.py[3w]?$"
    , confFileType = HumanReadable
    }


build :: Rules ()
build = do
  -- Proxy file for whole module
  buildDir id "/readme.md" %> \out -> do
    need $ fmap (buildDir id) (
      "/Dockerfile" :
      "/package.json" :
      "/yarn.lock" :
      "/.dockerignore" :

      "/source/config.json" :
      "/source/index.js" :
      "/source/main.js" :

      "/tests/fixtures/correct.py" :
      "/tests/fixtures/flawed.py" :
      "/tests/main.js" :
      [])

    writeFileChanged out "⚠ This directory contains only build files ⚠"

  buildDir id "/Dockerfile" %> \out -> do
    writeFileChanged out (makeDockerfile $ do
        runArgs ["apk", "add", "--no-cache", "python3"]
        runArgs ["pip3", "install", T.unpack $ "bandit==" <> toolVersion]
      )

  buildDir id "/source/*.js" %> \out -> do
    copyFileChanged ((sourceDir id "/source/") <> (takeFileName out)) out


  buildDir id "/source/config.json" %> \out -> do
    writeFileChanged out getConfigFile

  buildDir id "/package.json" %> \out -> do
    writeFileChanged out getPackageFile

  buildDir id "/.dockerignore" %> \out -> do
    writeFileChanged out "/node_modules/**\n"


  copyFilesToBuild id ["/tests/main.js"]

  buildDir id "/tests/fixtures/correct.py" %> \out -> do
    writeFileChanged out "print('test')\n"

  buildDir id "/tests/fixtures/flawed.py" %> \out -> do
    writeFileChanged out "\
      \def example1():\n\
      \    ####This is a long comment. \
        \This should be wrapped to fit within 72 characters.\n\
      \    some_tuple=(   1,2, 3,'a'  );\n\
      \    command_to_execute  = \"ls -lsa\" \n\
      \    subprocess.Popen(command_to_execute, shell=True)\n\
      \    return  some_tuple\n\
      \"
