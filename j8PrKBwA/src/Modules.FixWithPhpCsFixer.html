sourceDir :: String -> String
sourceDir path =
  "./source/tools/fix-with-php-cs-fixer" ++ path


buildDir :: String -> String
buildDir path =
  "./build/fix-with-php-cs-fixer" ++ path


copyFrom :: MonadFree EInstruction m
  => String
  -> (NonEmpty SourcePath)
  -> TargetPath
  -> m ()
copyFrom image sources dest =
  copyArgs sources dest Syntax.NoChown (Syntax.CopySource image)


getDockerfile :: String
getDockerfile = toDockerfileStr $ do
  let
    name = "ebekebe/ferambase"
    tag = "v1-11-gdf132ae_2017-11-02"
    alias = "builder"
    version = "2"
    urlDownload = "http://cs.sensiolabs.org/download"
    urlBinary = urlDownload ++ "/php-cs-fixer-v" ++ version ++ ".phar"

  from $ (tagged name tag) `aliased` alias

  workdir "/var/app"

  arg "SSH_KEY" Nothing
  runArgs ["configure-ssh", "$SSH_KEY"]

  copy ["package.json"] "package.json"
  copy ["yarn.lock"] "yarn.lock"
  copy ["build-patch"] "build-patch"

  runArgs ["yarn", "install"]

  ------------------------------

  from $ "stickee/docker-ci-node-php7" `tagged` "node-8"
  workdir "/var/app"

  copyFrom alias ["/var/app/node_modules"] "node_modules"

  run "apt-get update --yes"
  run "apt-get upgrade --yes"
  run $ "curl --location " ++ urlBinary ++ " --output php-cs-fixer"
  run "chmod a+x php-cs-fixer"
  run "mv php-cs-fixer /usr/local/bin/php-cs-fixer"

  -- Disable XDebug because it slows down php-cs-fixer
  run "phpdismod xdebug"

  copy ["yarn.lock"] "yarn.lock"
  copy ["package.json"] "package.json"
  copy ["tests"] "tests"
  copy ["source"] "source"

  runArgs ["yarn", "test"]

  -- TODO: Fix after https://github.com/hadolint/language-docker/issues/22
  entrypoint $ show (["yarn", "start", "--"] :: [String])

build :: Rules ()
build = do
  let
    dockerfile = buildDir "/Dockerfile"
    packgeJson = buildDir "/package.json"
    yarnLock = buildDir "/yarn.lock"
    dockerignore = buildDir "/.dockerignore"
    configJs = buildDir "/source/config.js"
    indexJs = buildDir "/source/index.js"
    mainJs = buildDir "/source/main.js"
    testJs = buildDir "/tests/main.js"

    correctPhp = buildDir "/tests/fixtures/correct.php"
    flawedPhp = buildDir "/tests/fixtures/flawed.php"
    invalidSyntaxPhp = buildDir "/tests/fixtures/invalid-syntax.php"


  -- Proxy file for whole module
  buildDir "/readme.md" %> \out -> do
    need (
      dockerfile :
      packgeJson :
      yarnLock :
      dockerignore :
      configJs :
      indexJs :
      mainJs :
      testJs :
      flawedPhp :
      invalidSyntaxPhp :
      correctPhp :
      [])
    writeFileChanged out "⚠ This directory contains only build files ⚠"


  packgeJson %> \out -> do
    writeFileChanged out getPackageFile


  buildDir "/source/*.js" %> \out -> do
    copyFileChanged ((sourceDir "/source/") ++ (takeFileName out)) out


  testJs %> \out -> do
    copyFileChanged (replace "build" "source/tools" out) out


  dockerfile %> \out -> do
    writeFileChanged out getDockerfile


  dockerignore %> \out -> do
    writeFileChanged out "/node_modules/**\n"


  flawedPhp %> \out -> do
    copyFileChanged (replace "build" "source/tools" out) out


  invalidSyntaxPhp %> \out -> do
    copyFileChanged (replace "build" "source/tools" out) out


  correctPhp %> \out -> do
    copyFileChanged (replace "build" "source/tools" out) out
