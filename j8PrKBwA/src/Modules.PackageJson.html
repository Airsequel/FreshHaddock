{-# LANGUAGE OverloadedStrings #-}
{-# LANGUAGE DeriveGeneric #-}

module Modules.PackageJson where

import GHC.Generics (Generic)
import Data.Aeson
import Data.Aeson.Encode.Pretty
import qualified Data.ByteString.Lazy.Char8 as LBS
import qualified Data.Text as T
import qualified Data.HashSet as HSet
import qualified Data.HashMap.Strict as HMap
import Data.Hashable
import Data.Function ((&))


{-| A person can be an author or a contributor. -}
data Person = Person
  { personName :: String
  , personEmail :: String
  , personUrl :: String
  } deriving (Eq, Show)

instance ToJSON Person where
  toJSON person = object
    [ "name" .= personName person
    , "email" .= personEmail person
    , "url" .= personUrl person
    ]


{-| Use one of the common licenses or a custom one -}
data License
  = UNLICENSED
  | MIT
  | GPL
  | ISC
  | License T.Text
  deriving (Eq, Show, Generic)

instance ToJSON License where
  toJSON (License license) = String license
  toJSON license = String (T.pack $ show license)



{-|
A JavaScript module the current one depends on.
If it's installed from npm `depUrl` is `Nothing`.
-}
data Dependency = Dependency
  { depName :: String
  , depVersion :: String
  , depUrl :: Maybe String
  } deriving (Eq, Show, Generic)

instance Hashable Dependency

getVersion :: Dependency -> String
getVersion dependency =
  case (depUrl dependency, depVersion dependency) of
    (Nothing, version) -> version
    (Just url, version) -> url ++ "#" ++ version

instance ToJSON Dependency where
  toJSON dependency = object
    [ (T.pack $ depName dependency) .= getVersion dependency]

dependencies ::
  [(String, String, Maybe String)] -> HSet.HashSet Dependency
dependencies =
  (HSet.fromList . (map (\(a,b,c) -> Dependency a b c)))


{-| Npm script which can be executed. -}
data Script = Script
  { scriptName :: String
  , scriptBody :: String
  } deriving (Eq, Show, Generic)

instance Hashable Script

scripts ::
  [(String, String)] -> HSet.HashSet Script
scripts =
  (HSet.fromList . (map (\(a,b) -> Script a b)))


data Package = Package
  { packName :: String
  , packVersion :: String
  , packDescription :: String
  , packMain :: String
  , packScripts :: HSet.HashSet Script
  , packKeywords :: [String]
  , packHomepage :: String
  , packLicense :: License
  , packPrivate :: Bool
  , packAuthor :: Person
  , packDependencies :: HSet.HashSet Dependency
  , packDevDependencies :: HSet.HashSet Dependency
  } deriving (Eq, Show)


getDepsMap
  :: (Package -> HSet.HashSet Dependency)
  -> Package
  -> HMap.HashMap String String
getDepsMap field package = package
  & field
  & HSet.map (\dep -> (depName dep, getVersion dep))
  & HSet.toList
  & HMap.fromList


instance ToJSON Package where
  toJSON package = object
    [ "name" .= packName package
    , "version" .= packVersion package
    , "description" .= packDescription package
    , "main" .= packMain package
    , "scripts" .= ( package
        & packScripts
        & HSet.map (\s -> (scriptName s, scriptBody s))
        & HSet.toList
        & HMap.fromList
        )
    , "keywords" .= packKeywords package
    , "homepage" .= packHomepage package
    , "license" .= packLicense package
    , "private" .= packPrivate package
    , "author" .= packAuthor package
    , "dependencies" .= getDepsMap packDependencies package
    , "devDependencies" .= getDepsMap packDevDependencies package
    -- TODO: Remove hardcoded "eslintConfig" field
    , "eslintConfig" .= object
      [ "extends" .= ("eslint-config-javascript" :: String) ]
    ]



examplePerson :: Person
examplePerson = Person "John" "john@example.org" "https://john.com"

exampleDependency :: Dependency
exampleDependency = Dependency "execa" "1.3.0" Nothing

exampleDevDependency :: Dependency
exampleDevDependency = Dependency "mocha" "1.6.0" Nothing

exampleScript :: Script
exampleScript = Script "test" "yarn test"

examplePackage :: Package
examplePackage = Package
  { packName = "test-module"
  , packVersion = "0.1.0"
  , packDescription = "Just a test module"
  , packMain = "test.js"
  , packScripts = HSet.singleton exampleScript
  , packKeywords = ["test", "module"]
  , packHomepage = "https://npmjs.com/packages/test-module"
  , packLicense = MIT
  , packPrivate = False
  , packAuthor = examplePerson
  , packDependencies = HSet.singleton exampleDependency
  , packDevDependencies = HSet.singleton exampleDevDependency
  }


showPretty :: Package -> String
showPretty value =
  LBS.unpack $ encodePretty' defConfig {confIndent = Spaces 2} value


example :: String
example =
  showPretty examplePackage
