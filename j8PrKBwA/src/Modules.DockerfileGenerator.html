{-# LANGUAGE FlexibleContexts #-}
{-# LANGUAGE NoImplicitPrelude #-}
{-# LANGUAGE OverloadedLists #-}
{-# LANGUAGE OverloadedStrings #-}


module Modules.DockerfileGenerator (makeDockerfile, makeDockerfileCustom)
where

import Protolude hiding (check, from)
import Control.Monad.Free.Class (MonadFree)
import Language.Docker
import Language.Docker.EDSL.Types (EInstruction)
import qualified Data.List.NonEmpty as NonEmpty
import qualified Language.Docker.EDSL as EDSL (interval, timeout, copyArgs)
import qualified Language.Docker.Syntax as Syntax


copyFrom :: MonadFree EInstruction m
  => [Char]
  -> (NonEmpty.NonEmpty Syntax.SourcePath)
  -> Syntax.TargetPath
  -> m ()
copyFrom image sources dest =
  EDSL.copyArgs sources dest Syntax.NoChown (Syntax.CopySource image)


makeDockerfileCustom :: EBaseImage -> EDockerfileM a -> [Char]
makeDockerfileCustom baseImage commands = toDockerfileStr $ do
  let
    name = "ebekebe/ferambase"
    tag = "v1-11-gdf132ae_2017-11-02"
    builder = "builder"
    healthCheckPort = 5000 :: Integer

  from $ (tagged name tag) `aliased` builder

  workdir "/var/app"

  arg "SSH_KEY" Nothing
  runArgs ["configure-ssh", "$SSH_KEY"]

  copy ["package.json"] "package.json"
  copy ["yarn.lock"] "yarn.lock"
  copy ["analytics"] "analytics"
  copy ["build-patch"] "build-patch"

  runArgs ["yarn", "install"]

  ------------------------------

  from baseImage
  workdir "/var/app"

  _ <- commands

  copyFrom builder ["/var/app/node_modules"] "node_modules"
  runArgs ["PATH=$PATH:/var/app/node_modules/.bin"]

  copy ["yarn.lock"] "yarn.lock"
  copy ["package.json"] "package.json"
  copy ["tests"] "tests"
  copy ["source"] "source"

  runArgs ["yarn", "test"]

  -- TODO: Improve after https://github.com/hadolint/language-docker/issues/22
  entrypoint $ show
    (["yarn", "start", "--health-check-port", show healthCheckPort] :: [[Char]])

  healthcheck $
    (check $ "curl -f http://localhost:" ++ show healthCheckPort
      ++ " > /dev/null || exit 1")
    `EDSL.interval` 10 {-sec-}
    `EDSL.timeout` 3 {-sec-}


makeDockerfile :: EDockerfileM a -> [Char]
makeDockerfile commands =
  makeDockerfileCustom "node:8-alpine" $ do
    runArgs ["apk", "add", "--no-cache", "curl"]
    runArgs ["curl", "--version"]
    commands
