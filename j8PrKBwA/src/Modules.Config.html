{-| Expected properties as defined in the signature -}
data Property = Property
  { propKey :: Text
  , propValue :: JsType
  } deriving (Eq, Show, Generic)

instance Hashable Property

properties :: [(Text, JsType)] -> HSet.HashSet Property
properties =
  (HSet.fromList . (fmap (\(key, value) -> Property key value)))


data RegexFlag
  = IgnoreCase
  | GlobalMatch
  | Multiline
  | Unicode
  | Sticky
  deriving (Eq, Show, Generic)


showRegexFlag :: RegexFlag -> Text
showRegexFlag flag = case flag of
  IgnoreCase -> "i"
  GlobalMatch -> "g"
  Multiline -> "m"
  Unicode -> "u"
  Sticky -> "y"


{-| JsRegex Flags Pattern  -}
data JsRegex
  = JsRegex [RegexFlag] Text
  | AnyPattern
  | NoPattern
  deriving (Eq, Show, Generic)

instance ToJSON JsRegex where
  toJSON (JsRegex flags pattern) = object
    [ "pattern" .= pattern
    , "flags" .= T.intercalate "" (fmap showRegexFlag flags)
    ]
  toJSON AnyPattern = object
    [ "pattern" .= ("." :: Text)
    , "flags" .= ("i" :: Text)
    ]
  toJSON NoPattern = object
    [ "pattern" .= ("" :: Text)
    , "flags" .= ("" :: Text)
    ]


data JsType
  = JsString | JsObject | JsPatchObjects
  deriving (Eq, Show, Generic)

instance Hashable JsType

instance ToJSON JsType where
  toJSON JsString = "string"
  toJSON JsObject = "object"
  toJSON JsPatchObjects = "patchObjects"


data ModuleType
  = FixingFunction | WarningFunction
  deriving (Eq, Show, Generic)

instance ToJSON ModuleType where
  toJSON FixingFunction = "fixing-function"
  toJSON WarningFunction = "warning-function"


data FileType
  = HumanReadable | Binary | AnyFileType
  deriving (Eq, Show, Generic)


getPropsMap
  :: (Signature -> HSet.HashSet Property)
  -> Signature
  -> HMap.HashMap Text JsType
getPropsMap property signature = signature
  & property
  & HSet.map (\(Property name theType) -> (name, theType))
  & HSet.toList
  & HMap.fromList


data Signature = Signature
  { sigType :: JsType
  , sigProperties :: HSet.HashSet Property
  , sigReturnType :: JsType
  } deriving (Eq, Show, Generic)

instance ToJSON Signature where
  toJSON signature = object
    [ "type" .= sigType signature
    , "properties" .= getPropsMap sigProperties signature
    , "returnType" .= sigReturnType signature
    ]


data ModuleConfig = ModuleConfig
  { confName :: Text
  , confVersion :: Text
  , confDescription :: Text
  , confType :: ModuleType
  , confSignature :: Signature
  , confTool :: Tool
  , confConfigFiles :: [Text]
  , confFilePattern :: JsRegex
  , confFileType :: FileType
  } deriving (Eq, Show, Generic)


instance ToJSON ModuleConfig where
  toJSON ModuleConfig {..} = object
    [ "name" .= confName
    , "version" .= confVersion
    , "description" .= confDescription
    , "type" .= confType
    , "signature" .= confSignature
    , "tool" .= confTool
    , "configFiles" .= confConfigFiles
    , "filePattern" .= confFilePattern
    ]


exampleConfig :: ModuleConfig
exampleConfig = ModuleConfig
  { confName = "test-config"
  , confVersion = "1.0.0"
  , confDescription = "Just a test config"
  , confType = FixingFunction
  , confSignature = Signature
    { sigType = JsObject
    , sigProperties = HSet.singleton (Property "repoPath" JsString)
    , sigReturnType = JsPatchObjects
    }
  , confTool = Tool
    { id = "test-tool"
    , name = "Test Tool"
    , version = "2.0.0"
    , description = "Just a test tool"
    , link = "https://test-tool.org"
    , languages = [Python]
    }
  , confConfigFiles = ["test-tool.yaml"]
  , confFilePattern = JsRegex [IgnoreCase] "\\.py[3w]?$"
  , confFileType = HumanReadable
  }


showPretty :: ModuleConfig -> [Char]
showPretty value =
  LBS.unpack $ (encodePretty' defConfig {confIndent = Spaces 2} value) <> "\n"
