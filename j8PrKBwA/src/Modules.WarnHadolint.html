{-# LANGUAGE OverloadedStrings #-}
{-# LANGUAGE OverloadedLists #-}
{-# LANGUAGE FlexibleContexts #-}


module Modules.WarnHadolint where

import Development.Shake
import Development.Shake.FilePath

import Language.Docker

import Modules.PackageJson
import Modules.PackageJsonDefault
import Modules.DockerfileGenerator


name :: String
name = "warn-hadolint"


sourceDir :: String -> String
sourceDir path =
  "./source/tools/" ++ name ++ path


buildDir :: String -> String
buildDir path =
  "./build/" ++ name ++ path


getPackageFile :: String
getPackageFile = do
  showPretty packageFileDefault
    { packName = "@feram/" ++ name
    , packVersion = "0.1.0"
    , packDescription = "Generate warnings with Hadolint"
    , packDependencies = dependencies (
        ( "@feram/api-client", "master",
          Just "git+ssh://git@github.com/feramhq/api-client") :
        ( "@feram/build-patch", "", Just "./build-patch") :
        ("bunyan", "^1.8.12", Nothing) :
        ("common-tags", "^1.7.2", Nothing) :
        ("diff", "^3.4.0", Nothing) :
        ("execa", "^0.8.0", Nothing) :
        ("fs-extra", "^5.0.0", Nothing) :
        [])
    }


copyFileToBuild :: String -> Rules ()
copyFileToBuild filePath = do
  buildDir filePath %> \out -> do
    copyFileChanged (sourceDir filePath) out


build :: Rules ()
build = do
  -- Proxy file for whole module
  buildDir "/readme.md" %> \out -> do
    need $ map buildDir (
      "/Dockerfile" :
      "/package.json" :
      "/yarn.lock" :
      "/.dockerignore" :

      "/source/config.js" :
      "/source/index.js" :
      "/source/main.js" :

      "/tests/fixtures/DockerfileCorrect" :
      "/tests/fixtures/DockerfileFlawed" :
      "/tests/main.js" :
      [])

    writeFileChanged out "⚠ This directory contains only build files ⚠"

  let
    downloadUrl = "https://github.com/hadolint/hadolint/releases/download\
      \/v1.5.1/hadolint-Linux-x86_64"

  buildDir "/Dockerfile" %> \out -> do
    writeFileChanged out (makeDockerfile $ do
        runArgs ["curl", "--location", downloadUrl, "--output", "hadolint"]
        runArgs ["chmod", "+x", "hadolint"]
        runArgs ["mv", "hadolint", "/usr/local/bin/hadolint"]
      )

  buildDir "/source/*.js" %> \out -> do
    copyFileChanged ((sourceDir "/source/") ++ (takeFileName out)) out

  buildDir "/package.json" %> \out -> do
    writeFileChanged out getPackageFile

  buildDir "/.dockerignore" %> \out -> do
    writeFileChanged out "/node_modules/**\n"

  copyFileToBuild "/tests/fixtures/DockerfileCorrect"
  copyFileToBuild "/tests/fixtures/DockerfileFlawed"
  copyFileToBuild "/tests/main.js"
