{-# LANGUAGE FlexibleContexts #-}
{-# LANGUAGE NoImplicitPrelude #-}
{-# LANGUAGE OverloadedLists #-}
{-# LANGUAGE OverloadedStrings #-}


module Modules.Helpers where

import Protolude
import Data.Text
import Development.Shake


sourceDir :: Text -> Text -> [Char]
sourceDir id path =
  unpack $ "./source/tools/" <> id <> path


buildDir :: Text -> Text -> [Char]
buildDir id path =
  unpack $ "./build/" <> id <> path


copyFilesToBuild :: Text -> [Text] -> Rules ()
copyFilesToBuild id filePaths = do
  sequence_ $ fmap
    (\filePath -> buildDir id filePath %> \out -> do
      copyFileChanged (sourceDir id filePath) out
    )
    filePaths
