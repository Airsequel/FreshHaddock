{- |
Description: - Generates an EcmaScript module "build-patch"
-}

module Modules.BuildPatch where

import Development.Shake
import Development.Shake.FilePath

import Modules.PackageJson


basePath :: String -> String
basePath subPath =
  "./build//build-patch" ++ subPath


build :: Rules ()
build = do
  let mainFile = "index.js"

  basePath "/package.json" %> \out -> do
    need [joinPath [dropFileName out, mainFile]]

    let
      package = Package
        { packName = "@feram/build-patch"
        , packVersion = "0.1.0"
        , packDescription = "Create patches for warnings"
        , packMain = mainFile
        , packScripts = scripts
          [ ("lint", "eslint --max-warnings=0 .")
          , ("unit-tests", "node tests/main.js")
          , ("test", "yarn lint && yarn unit-tests")
          ]
        , packKeywords = ["feram", "patch", "diff"]
        , packHomepage = "https://github.com/feramhq/modules"
        , packLicense = UNLICENSED
        , packPrivate = True
        , packAuthor =
            Person
              "Adrian Sieber"
              "mail@adriansieber.com"
              "https://adriansieber.com"
        , packDependencies = dependencies
          [ ("common-tags", "^1.7.2", Nothing) ]
        , packDevDependencies = dependencies
          [ ("eslint", "^4.8.0", Nothing)
          , ("eslint-config-javascript", "^1.3.0", Nothing)
          ]
        }
    writeFileChanged out $ showPretty package


  basePath ("/" ++ mainFile) %> \out -> do
    copyFileChanged ("source/build-patch/" ++ mainFile) out
