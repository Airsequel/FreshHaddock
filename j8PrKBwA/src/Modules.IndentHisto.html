{-# LANGUAGE FlexibleContexts #-}

{-# LANGUAGE NoImplicitPrelude #-}

{-# LANGUAGE OverloadedLists #-}

{-# LANGUAGE OverloadedStrings #-}





module Modules.IndentHisto where



import Protolude

import Unsafe

import qualified Data.Text as T





printError :: IO ()

printError = do

  file <- readFile "path/to/test/file.txt"

  putText "Error for 2 spaces:"

  print $ correlate 2 $ histogram $ fmap leadingSpace $ T.lines file

  putText "Error for 3 spaces:"

  print $ correlate 3 $ histogram $ fmap leadingSpace $ T.lines file

  putText "Error for 4 spaces:"

  print $ correlate 4 $ histogram $ fmap leadingSpace $ T.lines file

  putText "Error for 5 spaces:"

  print $ correlate 5 $ histogram $ fmap leadingSpace $ T.lines file

  putText "Error for 6 spaces:"

  print $ correlate 6 $ histogram $ fmap leadingSpace $ T.lines file

  putText "Error for 7 spaces:"

  print $ correlate 7 $ histogram $ fmap leadingSpace $ T.lines file

  putText "Error for 8 spaces:"

  print $ correlate 8 $ histogram $ fmap leadingSpace $ T.lines file





correlate :: Int -> [(Int, Int)] -> Int

correlate reference histo =

  let computeError (indentation, noOccurrence) =

        noOccurrence *

        (if indentation `mod` reference == 0

           then 0

           else 1)

  in sum $ fmap computeError histo





histogram :: Ord a => [a] -> [(a, Int)]

histogram =

  fmap (\x -> (unsafeHead x, length x)) . group . sort





leadingSpace :: Text -> Int

leadingSpace line =

  T.length $ T.takeWhile (== ' ') $ line
