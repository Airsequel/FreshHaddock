module Modules.IndentHisto where



import Data.List





printError :: IO ()

printError = do

  file <- readFile "path/to/test/file.txt"

  putStrLn "Error for 2 spaces:"

  putStrLn $ show $ correlate 2 $ histogram $ fmap leadingSpace $ lines file

  putStrLn "Error for 3 spaces:"

  putStrLn $ show $ correlate 3 $ histogram $ fmap leadingSpace $ lines file

  putStrLn "Error for 4 spaces:"

  putStrLn $ show $ correlate 4 $ histogram $ fmap leadingSpace $ lines file

  putStrLn "Error for 5 spaces:"

  putStrLn $ show $ correlate 5 $ histogram $ fmap leadingSpace $ lines file

  putStrLn "Error for 6 spaces:"

  putStrLn $ show $ correlate 6 $ histogram $ fmap leadingSpace $ lines file

  putStrLn "Error for 7 spaces:"

  putStrLn $ show $ correlate 7 $ histogram $ fmap leadingSpace $ lines file

  putStrLn "Error for 8 spaces:"

  putStrLn $ show $ correlate 8 $ histogram $ fmap leadingSpace $ lines file





correlate :: Int -> [(Int, Int)] -> Int

correlate reference histo =

  let computeError (indentation, noOccurrence) =

        noOccurrence *

        (if indentation `mod` reference == 0

           then 0

           else 1)

  in sum $ fmap computeError histo





histogram :: Ord a => [a] -> [(a, Int)]

histogram =

  fmap (\x -> (head x, length x)) . group . sort





leadingSpace :: String -> Int

leadingSpace line =

  length $ takeWhile (== ' ') $ line
