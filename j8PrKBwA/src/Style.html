{-# LANGUAGE OverloadedStrings #-}

module Style where

import Prelude hiding (div, span, filter)
import Clay hiding (wrap)
import Clay.Flexbox (wrap)


-- TODO: Fix floating point values (hsl, invert)
-- (blocked by https://github.com/sebastiaanvisser/clay/issues/153)


stylesheet :: Css
stylesheet = do
  let muted = hsl 0 0 70
  let bgMuted = hsl 0 0 20

  body ? do
    fontFamily [] [sansSerif]
    fontSize (px 14)
    color muted
    backgroundColor (hsl 0 0 10)

  h1 ? do
    fontSize (em 1.2)

  iframe ? do
    borderWidth 0

  div # ".row" ? do
    display flex
    flexFlow row wrap
    justifyContent flexStart

  div # ".widget" ? do
    backgroundColor bgMuted
    marginRight (em 1)
    marginBottom (em 1)
    let r = (px 2)
    borderRadius r r r r


  div # "#statusPageWrapper" ? do
    width (px 800)
    height (px 310)
    overflow hidden
    filter (invert 80)

  iframe # "#statusPage" ? do
    width (px 1000)
    height (px 610)
    position relative
    top  (px $ -300)
    left  (px $ -130)


  div # "#buildStatus" ? do
    padding (em 1) (em 2) (em 1) (em 2)


  div # "#codingHoursWrapper" ? do
    width (px 664)
    height (px 450)
    overflow hidden
    filter (invert 80)

  iframe # "#codingHours" ? do
    width (px 1220)
    height (px 1350)
    position relative
    top  (px $ -790)
    left  (px $ -500)


  div # "#starChart" ? do
    backgroundColor white
    filter (invert 80)
    img ? do
      width (em 40)
      height (em 22)


  div # "#burndownWrapper" ? do
    width (px (0.8 * 800))
    height (px 510)
    overflow hidden
    filter (invert 80)

  iframe # "#burndown" ? do
    width (px 800)
    height (px 1000)
    transform (scale 0.8 0.8)
    -- TODO: Replace with typed version
    -- Blocked by https://github.com/sebastiaanvisser/clay/issues/148
    "transform-origin" -: "0 0"
    position relative
    top (px $ -290)


  div # "#velocityWrapper" ? do
    width (px (0.8 * 1000))
    height (px 430)
    overflow hidden
    filter (invert 80)

  iframe # "#velocity" ? do
    width (px 1000)
    height (px 1000)
    transform (scale 0.8 0.8)
    -- TODO: Replace with typed version
    -- Blocked by https://github.com/sebastiaanvisser/clay/issues/148
    "transform-origin" -: "0 0"
    position relative
    top (px $ -150)


  table ? tr ? td # ":first-of-type" ? do
    textAlign end
    paddingRight  (em 1)
