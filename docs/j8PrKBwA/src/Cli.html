module Cli where

import Options.Applicative
import Data.Semigroup ((<>))

data Config = Config
  { user :: String
  , committer :: String
  , author :: String
  , shouldRunDry :: Bool
  , falsePositiveMaximum :: Float
  , search :: String
  , repoList :: String
  , submit :: Bool
  , reposPath :: String
  -- , logLevel :: String
  -- , logFilePath :: String
  , accessToken :: String
  }


sample :: Parser Config
sample = Config
  <$> strOption
    (  long "user"
    <> metavar "USER"
    <> help "GitHub user who initiated the check and sends the merge request"
    <> showDefault
    <> value "user@github"
    )
  <*> strOption
    (  long "committer"
    <> metavar "COMMITTER"
    <> help "The committer of the git commits"
    <> showDefault
    <> value "user@github"
    )
  <*> strOption
    (  long "author"
    <> metavar "AUTHOR"
    <> help "The author of the commits \
      \(not necessary as Feram is normally the author)"
    <> showDefault
    <> value "Feram <bot@feram.io>"
    )
  <*> switch
    (  long "should-run-dry"
    <> help "No fork, no pushing of changes and no merge request"
    )
  <*> option auto
    (  long "false-positive-maximum"
    <> metavar "VALUE"
    <> help "Maxium acceptable false positive value of generated patches"
    <> showDefault
    <> value 1
    )
  <*> strOption
    (  long "search"
    <> metavar "SEARCH_STRING"
    <> help "Search-string to find repos. One is selected randomly and checked."
    <> showDefault
    <> value "*"
    )
  <*> strOption
    (  long "repo-list"
    <> metavar "REPO_LIST"
    <> help "Comma separated list of repos to check"
    <> showDefault
    <> value "*"
    )
  <*> switch
    (  long "submit"
    <> help "Push already created local fixes and create merge request for them"
    )
  <*> strOption
    (  long "repos-path"
    <> metavar "REPOS_PATH"
    <> help "Path to save the repositories"
    <> showDefault
    <> value "~/.feram/repos"
    )
  <*> strOption
    (  long "access-token"
    <> metavar "ACCESS_TOKEN"
    <> help "Token to access the API of code hosting providers"
    <> showDefault
    <> value ""
    )


main :: IO ()
main =
  let
    opts = info (sample <**> helper)
      ( fullDesc
     <> progDesc "Print a greeting for TARGET"
     <> header "user - a test for optparse-applicative" )
  in
    startFeram =<< execParser opts


startFeram :: Config -> IO ()
startFeram (Config user committer author shouldRunDry falsePositiveMaximum
    search
    repoList
    submit
    reposPath
    accessToken
  ) = putStrLn $ "Hello, " ++ user
